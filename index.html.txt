<script>
    const SERVER_URL = "https://your-server.example.com/api/points";

    const map = L.map("map").setView([51.1, 69.3], 6);

    L.tileLayer("https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png", {
        attribution: "© OpenStreetMap"
    }).addTo(map);

    async function loadData() {
        try {
            const response = await fetch(SERVER_URL);
            const points = await response.json();

            updateStats(points);
            drawMarkers(points);

        } catch (e) {
            console.error("Ошибка загрузки данных", e);
        }
    }

    function updateStats(points) {
        document.getElementById("total-points").textContent = points.length;

        const avgIndex = (
            points.reduce((s, p) => s + p.pollutionIndex, 0) / points.length
        ).toFixed(2);

        document.getElementById("avg-index").textContent = avgIndex;

        document.getElementById("critical-count").textContent =
            points.filter(p => p.pollutionIndex >= 7.5).length;
    }

    function drawMarkers(points) {
        points.forEach(p => {
            L.circleMarker([p.lat, p.lng], {
                radius: 8,
                color: "#000",
                weight: 1,
                fillColor: getColor(p.pollutionIndex),
                fillOpacity: 0.85
            })
            .addTo(map)
            .bindPopup(`
                <strong>${p.name}</strong><br>
                Индекс загрязнения: ${p.pollutionIndex}
            `);
        });
    }

    function getColor(value) {
        if (value <= 2.5) return "green";
        if (value <= 5) return "yellow";
        if (value <= 7.5) return "orange";
        return "red";
    }

    loadData();
</script>
